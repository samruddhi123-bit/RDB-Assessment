SELECT * FROM users;

SELECT * FROM progress;
	
SELECT email_domain FROM users WHERE email_domain LIKE '%.edu' GROUP BY email_domain LIMIT 25;

SELECT COUNT(*) FROM users WHERE email_domain LIKE '%.edu' AND city = 'New York';

SELECT COUNT(*) FROM users WHERE mobile_app = 'mobile-user';

SELECT sign_up_at, strftime('%S', sign_up_at) FROM users GROUP BY 1 LIMIT 20;


SELECT 
  strftime('%H', sign_up_at) AS hour, 
  COUNT(*) AS sign_up_count
FROM users
GROUP BY hour
ORDER BY hour;


SELECT * 
FROM users 
JOIN progress 
ON users.user_id = progress.user_id;


SELECT u.email_domain, 
p.learn_cpp, 
p.learn_html, 
p.learn_java, 
learn_javascript, 
learn_sql 
FROM users u JOIN progress p ON u.user_id = p.user_id 
WHERE u.email_domain LIKE '%.edu' 
GROUP BY u.email_domain 
ORDER BY u.email_domain;


SELECT u.user_id, 
p.learn_cpp, 
p.learn_sql, 
p.learn_html, 
p.learn_javascript, 
p.learn_java 
FROM users u JOIN progress p ON u.user_id = p.user_id 
WHERE u.city = 'New York';


SELECT u.user_id,
       p.learn_cpp,
       p.learn_sql,
       p.learn_html,
       p.learn_javascript,
       p.learn_java
FROM users u
JOIN progress p ON u.user_id = p.user_id
WHERE u.city = 'Chicago';
